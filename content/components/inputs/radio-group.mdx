## Usage

```jsx
import { RadioGroup, RadioGroupITem } from "@/components/core/radio-group";
```

```jsx
<RadioGroup defaultValue="option-one">
  <RadioGroupItem value="option-one">Option One</>
  <RadioGroupItem value="option-two">Option Two</>
</RadioGroup>
```
