## Usage

```jsx
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuLabel,
  DropdownMenuSeparator,
  DropdownMenuTrigger,
} from "@/components/ui/dropdown-menu";

const Menu = () => (
  <MenuRoot>
    <MenuTrigger>Open</MenuTrigger>
    <MenuContent>
      <MenuLabel>My Account</MenuLabel>
      <MenuSeparator />
      <MenuItem>Profile</MenuItem>
      <MenuItem>Billing</MenuItem>
    </MenuContent>
  </MenuRoot>
);
```
